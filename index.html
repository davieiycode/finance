function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const timezone = ss.getSpreadsheetTimeZone();
  const sheets = ss.getSheets();
  let sheet = null;
  for (let s of sheets) {
    if (s.getName().toLowerCase() === 'transaction') { sheet = s; break; }
  }

  if (!sheet) return ContentService.createTextOutput(JSON.stringify({ status: 'error', transactions: [] })).setMimeType(ContentService.MimeType.JSON);
  const data = sheet.getDataRange().getValues(); 
  if (data.length < 2) return ContentService.createTextOutput(JSON.stringify({ status: 'success', transactions: [] })).setMimeType(ContentService.MimeType.JSON);
  
  const headers = data.shift();
  const transactions = data.map(row => {
    let t = {};
    headers.forEach((h, i) => {
      let v = row[i];
      let hName = h.toString().trim().toLowerCase();
      
      // LOGIKA ANTI 1899 UNTUK TANGGAL
      if (hName === 'date') {
        if (v instanceof Date) {
          if (v.getFullYear() < 1920) {
            v = Utilities.formatDate(new Date(), timezone, "yyyy-MM-dd"); 
          } else {
            v = Utilities.formatDate(v, timezone, "yyyy-MM-dd");
          }
        }
        else if (typeof v === 'string' && v.includes('T')) v = v.split('T')[0];
        t.date = v;
      }
      else if (hName === 'time') {
        if (v instanceof Date) v = Utilities.formatDate(v, timezone, "HH:mm");
        t.time = v ? v.toString() : '';
      }
      
      // DATA UTAMA (Author, Description, Receipt, dll)
      else if (hName === 'transaction id') t.id = v;
      else if (hName === 'merchant') t.merchant = v;
      else if (hName === 'item name' || hName === 'name') t.name = v;
      else if (hName === 'category') t.category = v;
      else if (hName === 'category group') t.categoryGroup = v;
      else if (hName === 'type') t.type = v;
      else if (hName === 'description') t.description = v;
      else if (hName === 'author') t.author = v;
      else if (hName === 'receipt') t.receipt = v;
      else if (hName === 'cleared') t.cleared = (v === 'Yes');
      
      // ANALISIS (Tags & Project)
      else if (hName === 'analytics tags' || hName === 'tags') t.tags = v ? v.toString().split(', ') : [];
      else if (hName === 'associated project' || hName === 'project') t.projects = v ? v.toString().split(', ') : [];
      
      // ANGKA & AKUN (Dipisah dengan benar)
      else if (hName === 'amount (per unit)' || hName === 'amount') {
        t.amount = parseFloat(v.toString().replace(/[^0-9.-]/g, '')) || 0;
      }
      else if (hName === 'total') {
        t.total = parseFloat(v.toString().replace(/[^0-9.-]/g, '')) || 0;
      }
      else if (hName === 'discount') t.discount = parseFloat(v) || 0;
      else if (hName === 'additional fee') t.fee = parseFloat(v) || 0;
      else if (hName === 'payment source account' || hName === 'payment account') t.accountPayment = v;
      else if (hName === 'beneficiary account' || hName === 'received account') t.accountReceived = v;
      
      // LAINNYA
      else if (hName === 'currency') t.currency = v;
      else if (hName === 'quantity') t.qty = parseFloat(v) || 1;
      else if (hName === 'unit scale' || hName === 'scale') t.scale = v;
      else if (hName === 'exchange rate') t.exchangeRate = parseFloat(v) || 1;
      
      // METADATA
      else if (hName === 'budget amount') t.budgetAmount = v;
      else if (hName === 'budget period') t.budgetPeriod = v;
      else if (hName === 'flow') t.flow = v;
      else if (hName === 'year') t.year = v;
      else if (hName === 'month') t.month = v;
      else if (hName === 'update time') t.updateTime = v;
      else if (hName === 'xp') t.xp = v;
    });
    return t;
  });
  
  return ContentService.createTextOutput(JSON.stringify({ status: 'success', transactions: transactions }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    const contents = JSON.parse(e.postData.contents);
    const transactions = contents.transactions || [];
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName('transaction');
    if (!sheet) { sheet = ss.insertSheet('transaction'); } 
    
    // Clear content but keep header row 1 formatting/frozen if possible
    if (sheet.getLastRow() > 0) {
      sheet.getRange(1, 1, sheet.getLastRow(), 31).clearContent();
    }
    
    const headers = [
      'Transaction ID', 'Date', 'Time', 'Category', 'Category Group', 
      'Merchant', 'Item Name', 'Description', 'Amount (per Unit)', 'Currency', 
      'Exchange Rate', 'Quantity', 'Unit Scale', 'Type', 'Cleared', 
      'Payment Source Account', 'Beneficiary Account', 'Receipt', 'Analytics Tags', 'Associated Project', 
      'Author', 'Discount', 'Additional Fee', 'Total', 'Budget Amount', 
      'Budget Period', 'Flow', 'Year', 'Month', 'Update Time', 'XP'
    ];
    
    const rows = [headers];
    
    if (transactions.length > 0) {
      transactions.forEach(t => {
        const d = new Date(t.date || new Date());
        rows.push([
          t.id, t.date, t.time || '', t.category, t.categoryGroup || '', t.merchant, t.name, t.description || '',
          t.amount || 0, t.currency || 'IDR', t.exchangeRate || 1, t.qty || 1, t.scale || 'pcs',
          t.type, t.cleared ? 'Yes' : 'No', t.accountPayment || '', t.accountReceived || '',
          t.receipt || '', (t.tags || []).join(', '), (t.projects || []).join(', '), t.author || '',
          t.discount || 0, t.fee || 0, t.total || 0, t.budgetAmount || 0, t.budgetPeriod || 'Monthly', 
          t.flow || (t.type === 'Income' ? 'In' : 'Out'), t.year || d.getFullYear(), t.month || (d.getMonth() + 1), t.updateTime || new Date().toISOString(), t.xp || 0
        ]);
      });
    }
    
    sheet.getRange(1, 1, rows.length, 31).setValues(rows);
    sheet.getRange(1, 1, 1, 31).setFontWeight('bold').setBackground('#f3f3f3');
    sheet.setFrozenRows(1);
    
    return ContentService.createTextOutput(JSON.stringify({ status: 'success' })).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: err.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}
