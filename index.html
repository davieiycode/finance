This script allows the app to communicate securely with your personal spreadsheet.

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheets = ss.getSheets();
  const timezone = ss.getSpreadsheetTimeZone();
  const response = { status: 'success' };

  sheets.forEach(sh => {
    const name = sh.getName().toLowerCase().trim();
    const data = sh.getDataRange().getValues();
    if (data.length < 2) { response[name + 's'] = []; return; }

    const headers = data.shift();
    const records = data.map(row => {
      let r = {};
      headers.forEach((h, i) => {
        let v = row[i];
        let hName = (h || '').toString().trim().toLowerCase();
        if (!hName) return;

        // Common Field Mapping for all sheets
        if (hName.includes('id')) r.id = v;
        else if (hName === 'name' || hName === 'merchant' || hName === 'account' || hName === 'item') r.name = v;
        
        // Transaction Specific Mapping
        if (name === 'transaction') {
          if (hName === 'date') {
            if (v instanceof Date) v = Utilities.formatDate(v, timezone, "yyyy-MM-dd");
            else if (typeof v === 'string' && v.includes('T')) v = v.split('T')[0];
            r.date = v;
          }
          else if (hName === 'time') {
            if (v instanceof Date) v = Utilities.formatDate(v, timezone, "HH:mm");
            r.time = v ? v.toString() : '';
          }
          else if (hName === 'merchant') r.merchant = v;
          else if (hName === 'item name' || hName === 'name') r.name = v;
          else if (hName === 'category') r.category = v;
          else if (hName === 'category group') r.categoryGroup = v;
          else if (hName === 'type') r.type = v;
          else if (hName === 'description') r.description = v;
          else if (hName === 'author') r.author = v;
          else if (hName === 'receipt') r.receipt = v;
          else if (hName === 'cleared') r.cleared = (v === 'Yes');
          else if (hName === 'analytics tags' || hName === 'tags') r.tags = v ? v.toString().split(', ') : [];
          else if (hName === 'associated project' || hName === 'project') r.projects = v ? v.toString().split(', ') : [];
          else if (hName === 'amount (per unit)' || hName === 'amount') r.amount = parseFloat(v) || 0;
          else if (hName === 'total') r.total = parseFloat(v) || 0;
          else if (hName === 'discount') r.discount = parseFloat(v) || 0;
          else if (hName === 'additional fee') r.fee = parseFloat(v) || 0;
          else if (hName === 'payment source account' || hName === 'payment account') r.accountPayment = v;
          else if (hName === 'beneficiary account' || hName === 'received account') r.accountReceived = v;
          else if (hName === 'currency') r.currency = v;
          else if (hName === 'quantity') r.qty = parseFloat(v) || 1;
          else if (hName === 'unit scale' || hName === 'scale') r.scale = v;
          else if (hName === 'exchange rate') r.exchangeRate = parseFloat(v) || 1;
          else if (hName === 'xp') r.xp = v;
        } 
        // Account/Master Data Specific (Generic headers)
        else {
          r[h.toString().trim()] = v;
          // Normalized keys (for app processing)
          if (hName === 'balance' || hName === 'limit' || hName === 'target' || hName === 'saved' || hName === 'price') {
             r[hName] = parseFloat(v) || 0;
          }
          if (hName === 'manufacturer' || hName === 'manu') r.manu = v;
          if (hName === 'model') r.model = v;
          if (hName === 'sku' || hName === 'barcode') r.sku = v;
          if (hName === 'notes' || hName === 'description') r.notes = v;
          if (hName === 'expiry' || hName === 'expirydate' || hName === 'cardexpiry') r.expiry = v;
          if (hName === 'cardexpiry') r.cardExpiry = v;
          if (hName === 'since') r.since = v;
          if (hName === 'provider') r.provider = v;
        }
      });
      return r;
    });

    // Match app's expected keys
    let finalKey = name + 's'; 
    if (name === 'transaction') finalKey = 'transactions';
    if (name === 'membership') finalKey = 'membership_cards';
    response[finalKey] = records;
  });

  return ContentService.createTextOutput(JSON.stringify(response)).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    const contents = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    
    // Process each entity type
    const entities = [
      { key: 'transactions', sheetName: 'transaction', headers: [
        'Transaction ID', 'Date', 'Time', 'Category', 'Category Group', 
        'Merchant', 'Item Name', 'Description', 'Amount (per Unit)', 'Currency', 
        'Exchange Rate', 'Quantity', 'Unit Scale', 'Type', 'Cleared', 
        'Payment Source Account', 'Beneficiary Account', 'Receipt', 'Analytics Tags', 'Associated Project', 
        'Author', 'Discount', 'Additional Fee', 'Total', 'Update Time', 'XP'
      ]},
      { key: 'accounts', sheetName: 'account', headers: ['ID', 'Name', 'Type', 'Provider', 'Card Expiry', 'Balance', 'Status', 'Color'] },
      { key: 'merchants', sheetName: 'merchant', headers: ['ID', 'Name', 'Type'] },
      { key: 'items', sheetName: 'item', headers: ['ID', 'Name', 'Category', 'Manufacturer', 'Model', 'SKU', 'Price', 'Unit', 'Expiry Date', 'Notes', 'Status'] },
      { key: 'budgets', sheetName: 'budget', headers: ['ID', 'Name', 'Type', 'Limit', 'Period'] },
      { key: 'goals', sheetName: 'goal', headers: ['ID', 'Name', 'Target', 'Saved', 'Deadline'] },
      { key: 'membership_cards', sheetName: 'membership', headers: ['ID', 'Name', 'Code', 'Expiry', 'Since', 'Notes', 'Type', 'Color'] }
    ];

    entities.forEach(ent => {
      const data = contents[ent.key];
      if (!data) return; // Skip if no data for this sheet
      
      let sheet = ss.getSheetByName(ent.sheetName);
      if (!sheet) { sheet = ss.insertSheet(ent.sheetName); }
      
      const rows = [ent.headers];
      data.forEach(item => {
        const row = ent.headers.map(h => {
          const hName = h.toLowerCase().replace(/\s+/g, '');
          // Specific mappings for complex types
          if (h === 'Analytics Tags' || h === 'Tags') return (item.tags || []).join(', ');
          if (h === 'Associated Project' || h === 'Project') return (item.projects || []).join(', ');
          if (h === 'Cleared') return item.cleared ? 'Yes' : 'No';
          
          // Generic match by property name logic
          for (let k in item) {
             if (k.toLowerCase().replace(/\s+/g, '') === hName) return item[k];
          }
          // Specific ID fallback
          if (h === 'ID' || h === 'Transaction ID') return item.id;
          if (h === 'Name') return item.name;
          
          // Re-map common finance keys
          if (h === 'Payment Source Account') return item.accountPayment || '';
          if (h === 'Beneficiary Account') return item.accountReceived || '';
          if (h === 'Amount (per Unit)') return item.amount || 0;
          if (h === 'Manufacturer') return item.manu || '';
          if (h === 'SKU') return item.sku || '';
          if (h === 'Expiry Date' || h === 'Card Expiry') return item.expiry || item.cardExpiry || '';
          
          return item[h] || ''; // Fallback to raw property
        });
        rows.push(row);
      });

      sheet.getRange(1, 1, sheet.getLastRow() || 1, ent.headers.length).clearContent();
      sheet.getRange(1, 1, rows.length, ent.headers.length).setValues(rows);
      sheet.getRange(1, 1, 1, ent.headers.length).setFontWeight('bold').setBackground('#f3f3f3');
      sheet.setFrozenRows(1);
    });

    return ContentService.createTextOutput(JSON.stringify({ status: 'success' })).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: err.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}
